46      /**
47       * Creates instance with the <i>mojo</i> to use and the <i>spec</i> to write to.
48       *
49       * @param mojo The mojo with the attributes to generate a spec file for.
50       * @param spec The target to write the spec file to.

52      public SpecWriter( AbstractRPMMojo mojo, PrintWriter spec )
53      {
54          super();
55          this.mojo = mojo;
56          this.spec = spec;
57      }

59      /**
60       * Writes the spec file to <i>spec</i> using the attributes of <i>mojo</i>.
61       *
62       * @throws MojoExecutionException
63       * @throws IOException
64       */
65      public void writeSpecFile()
66          throws MojoExecutionException, IOException
67      {
68          writeList( mojo.getDefineStatements(), "%define " );
69  
70          spec.println( "Name: " + mojo.getName() );
71          spec.println( "Version: " + mojo.getVersion() );
72          spec.println( "Release: " + mojo.getRelease() );
73  
74          writeNonNullDirective( "Summary", mojo.getSummary() );
75          writeNonNullDirective( "License", mojo.getLicense() );
76          writeNonNullDirective( "Distribution", mojo.getDistribution() );
77  
78          if ( mojo.getEpoch() != null )
79          {
80              spec.println( "Epoch: " + mojo.getEpoch() );
81          }
82  
83          if ( mojo.getIcon() != null )
84          {
85              spec.println( "Icon: " + mojo.getIcon().getName() );
86          }
87  
88          writeNonNullDirective( "Vendor", mojo.getVendor() );
89          writeNonNullDirective( "URL", mojo.getUrl() );
90          writeNonNullDirective( "Group", mojo.getGroup() );
91          writeNonNullDirective( "Packager", mojo.getPackager() );
92  
93          writeList( mojo.getProvides(), "Provides: " );
94          writeList( mojo.getRequires(), "Requires: " );
95          writeList( mojo.getPrereqs(), "PreReq: " );
96          writeList( mojo.getObsoletes(), "Obsoletes: " );
97          writeList( mojo.getConflicts(), "Conflicts: " );
98  
99          spec.println( "autoprov: " + ( mojo.isAutoProvides() ? "yes" : "no" ) );
100         spec.println( "autoreq: " + ( mojo.isAutoRequires() ? "yes" : "no" ) );
101 
102         if ( mojo.getPrefixes() != null )
103         {
104             for ( String prefix : mojo.getPrefixes() )
105             {
106                 spec.println( "Prefix: " + prefix );
107             }
108         }
109 
110         if ( "noarch".equals( mojo.getTargetArch() ) )
111         {
112             spec.println( "BuildArch: " + mojo.getTargetArch() );
113         }
114         spec.println( "BuildRoot: " + FileHelper.toUnixPath( mojo.getRPMBuildroot() ) );
115         spec.println();
116         spec.println( "%description" );
117         if ( mojo.getDescription() != null )
118         {
119             spec.println( mojo.getDescription() );
120         }
121 
122         if ( !mojo.getMappings().isEmpty() )
123         {
124             writeMove();
125         }
126 
127         writeLinks();
128 
129         if ( mojo.getInstallScriptlet() != null )
130         {
131             spec.println();
132             mojo.getInstallScriptlet().writeContent( spec );
133         }
134 
135         writeFiles();
136 
137         writeScripts();
138 
139         if ( mojo.getTriggers() != null )
140         {
141             for ( BaseTrigger trigger : mojo.getTriggers() )
142             {
143                 trigger.writeTrigger( spec );
144             }
145         }
146 
147         if ( mojo.getChangelog() != null )
148         {
149             spec.println();
150             spec.println( "%changelog" );
151             spec.println( mojo.getChangelog() );
152         }
153     }

155     /**
156      * Writes the %files directive based on {@link AbstractRPMMojo#mappings}.
157      */
158     private void writeFiles()
159     {
160         final Log log = mojo.getLog();
161 
162         spec.println();
163         spec.println( "%files" );
164         spec.println( getDefAttrString() );
165 
166         for ( Mapping map : mojo.getMappings() )
167         {
168             // For each mapping we need to determine which files in the destination were defined by this
169             // mapping so that we can write the %attr statement correctly.
170 
171             final String destination = map.getDestination();
172             final File absoluteDestination = map.getAbsoluteDestination();
173 
174             final String attrString =
175                 map.getAttrString( mojo.getDefaultFilemode(), mojo.getDefaultGroupname(), mojo.getDefaultUsername() );
176             final String baseFileString = attrString + "  \"" + destination + FileHelper.UNIX_FILE_SEPARATOR;
177 
178             if ( map.hasSoftLinks() && !absoluteDestination.exists() )
179             {
180                 // @TODO will this ever happen since absoluteDestination.exists() always likely true
181                 log.debug( "writing attribute string for directory created by soft link: " + destination );
182 
183                 spec.println( attrString + " \"" + destination + "\"" );
184 
185                 continue;
186             }
187 
188             final List<String> links = map.getLinkedFileNamesRelativeToDestination();
189 
190             if ( map.isSoftLinkOnly() )
191             {
192                 // map has only soft links, no need to do the scan (MRPM-173)
193                 log.debug( "writing attribute string for softlink only source" );
194                 for ( String link : links )
195                 {
196                     spec.print( baseFileString );
197                     spec.println( link + "\"" );
198                 }
199                 continue;
200             }
201 
202             // special handling for single map source directory with lots of files such UI/Nodejs
203             // without the need for slow scanner which can be up 10 of minutes for large set
204             if ( map.isSourceDirsOnly() && map.isDirectoryIncluded() && !map.isRecurseDirectories() )
205             {
206                 boolean simpleDir = true;
207                 for ( Source source : map.getSources() )
208                 {
209                     if ( !source.isSingleDir() )
210                     {
211                         simpleDir = false;
212                         break;
213                     }
214                 }
215 
216                 if ( simpleDir )
217                 {
218                     log.debug( "writing attribute string for directory: " + destination );
219                     spec.println( attrString + " \"" + destination + "\"" );
220                     continue;
221                 }
222             }
223 
224             final List<String> includes = map.getCopiedFileNamesRelativeToDestination();
225 
226             log.debug( "scanning: " + absoluteDestination );
227             final DirectoryScanner scanner = new DirectoryScanner();
228             scanner.setBasedir( absoluteDestination );
229 
230             // the linked files are not present yet (will be "installed" during rpm build)
231             // so they cannot be "included"
232             scanner.setIncludes( includes.isEmpty() ? null : includes.toArray( new String[includes.size()] ) );
233             scanner.setExcludes( null );
234             scanner.scan();
235 
236             if ( scanner.isEverythingIncluded() && links.isEmpty() && map.isDirectoryIncluded()
237                 && !map.isRecurseDirectories() )
238             {
239                 log.debug( "writing attribute string for directory: " + destination );
240                 spec.println( attrString + " \"" + destination + "\"" );
241             }
242             else
243             {
244                 log.debug( "writing attribute string for identified files in directory: " + destination );
245 
246                 // only list files if requested (directoryIncluded == false) or we have to
247                 if ( !( map.isDirectoryIncluded() && scanner.isEverythingIncluded() && links.isEmpty() ) )
248                 {
249                     final String[] files = scanner.getIncludedFiles();
250 
251                     for ( String file : files )
252                     {
253                         spec.print( baseFileString );
254                         spec.println( StringUtils.replace( file, "\\", "/" ) + "\"" );
255                     }
256                 }
257 
258                 if ( map.isRecurseDirectories() )
259                 {
260                     final String[] dirs = scanner.getIncludedDirectories();
261 
262                     if ( map.isDirectoryIncluded() )
263                     {
264                         // write out destination first
265                         spec.println( baseFileString + "\"" );
266                     }
267 
268                     for ( String dir : dirs )
269                     {
270                         // do not write out base file (destination) again
271                         if ( dir.length() > 0 )
272                         {
273                             spec.print( baseFileString );
274                             spec.println( dir + "\"" );
275                         }
276                     }
277                 }
278 
279                 // since the linked files are not present in directory (yet), the scanner will not find them
280                 for ( String link : links )
281                 {
282                     spec.print( baseFileString );
283                     spec.println( link + "\"" );
284                 }
285             }
286         }
287     }

289     /**
290      * Writes the beginning of the <i>%install</i> which includes moving all files from the
291      * {@link AbstractRPMMojo#getBuildroot()} to {@link AbstractRPMMojo#getRPMBuildroot()}.
292      */
293     private void writeMove()
294     {
295         final String tmpBuildRoot = FileHelper.toUnixPath( mojo.getBuildroot() );
296         spec.println();
297         spec.println( "%install" );
298         spec.println();
299         spec.println( "if [ -d $RPM_BUILD_ROOT ];" );
300         spec.println( "then" );
301         spec.print( "  mv " );
302         spec.print( tmpBuildRoot );
303         spec.println( "/* $RPM_BUILD_ROOT" );
304         spec.println( "else" );
305         spec.print( "  mv " );
306         spec.print( tmpBuildRoot );
307         spec.println( " $RPM_BUILD_ROOT" );
308         spec.println( "fi" );
309     }

311     /**
312      * Writes the install commands to link files.
313      */
314     private void writeLinks()
315     {
316         if ( !mojo.getLinkTargetToSources().isEmpty() )
317         {
318             spec.println();
319 
320             for ( Map.Entry<String, List<SoftlinkSource>> directoryToSourcesEntry : mojo.getLinkTargetToSources().entrySet() )
321             {
322                 String directory = directoryToSourcesEntry.getKey();
323                 if ( directory.startsWith( "/" ) )
324                 {
325                     directory = directory.substring( 1 );
326                 }
327                 if ( directory.endsWith( "/" ) )
328                 {
329                     directory = directory.substring( 0, directory.length() - 1 );
330                 }
331 
332                 final List<SoftlinkSource> sources = directoryToSourcesEntry.getValue();
333                 final int sourceCnt = sources.size();
334 
335                 if ( sourceCnt == 1 )
336                 {
337                     final SoftlinkSource linkSource = sources.get( 0 );
338 
339                     final String macroEvaluatedLocation = linkSource.getMacroEvaluatedLocation();
340 
341                     final File buildSourceLocation;
342                     if ( macroEvaluatedLocation.startsWith( "/" ) )
343                     {
344                         buildSourceLocation = new File( mojo.getBuildroot(), macroEvaluatedLocation );
345                     }
346                     else
347                     {
348                         buildSourceLocation = new File( mojo.getBuildroot(), directory + '/' + macroEvaluatedLocation );
349                     }
350 
351                     if ( buildSourceLocation.isDirectory() )
352                     {
353                         final DirectoryScanner scanner = scanLinkSource( linkSource, buildSourceLocation );
354 
355                         if ( scanner.isEverythingIncluded() )
356                         {
357                             final File destinationFile = linkSource.getSourceMapping().getAbsoluteDestination();
358                             destinationFile.delete();
359 
360                             spec.print( "ln -s " );
361                             spec.print( linkSource.getLocation() );
362                             spec.print( " $RPM_BUILD_ROOT/" );
363                             spec.print( directory );
364 
365                             final String dest = linkSource.getDestination();
366                             if ( dest != null )
367                             {
368                                 spec.print( '/' );
369                                 spec.print( dest );
370                                 linkSource.getSourceMapping().addLinkedFileNameRelativeToDestination( dest );
371                             }
372 
373                             spec.println();
374                         }
375                         else
376                         {
377                             linkScannedFiles( directory, linkSource, scanner );
378                         }
379                     }
380                     else
381                     {
382                         linkSingleFile( directory, linkSource );
383                     }
384                 }
385                 else
386                 {
387                     for ( SoftlinkSource linkSource : sources )
388                     {
389                         final String sourceLocation = linkSource.getMacroEvaluatedLocation();
390                         final File buildSourceLocation;
391                         if ( sourceLocation.startsWith( "/" ) )
392                         {
393                             buildSourceLocation = new File( mojo.getBuildroot(), sourceLocation );
394                         }
395                         else
396                         {
397                             buildSourceLocation = new File( mojo.getBuildroot(), directory + '/' + sourceLocation );
398                         }
399 
400                         if ( buildSourceLocation.isDirectory() )
401                         {
402                             final DirectoryScanner scanner = scanLinkSource( linkSource, buildSourceLocation );
403 
404                             linkScannedFiles( directory, linkSource, scanner );
405                         }
406                         else
407                         {
408                             linkSingleFile( directory, linkSource );
409                         }
410                     }
411                 }
412             }
413         }
414     }

416     /**
417      * Writes soft link from <i>linkSource</i> to <i>directory</i> for all files in the <i>scanner</i>.
418      *
419      * @param directory Directory to link to.
420      * @param linkSource Source to link from. {@link SoftlinkSource#getLocation()} must be a {@link File#isDirectory()
421      *            directory}.
422      * @param scanner Scanner used to scan the {@link SoftlinkSource#getLocation() linSource location}.
423      */
424     private void linkScannedFiles( String directory, final SoftlinkSource linkSource, final DirectoryScanner scanner )
425     {
426         final String[] files = scanner.getIncludedFiles();
427         final String sourceLocation = linkSource.getLocation();
428 
429         final String targetPrefix = sourceLocation + FileHelper.UNIX_FILE_SEPARATOR;
430         final String sourcePrefix = directory + FileHelper.UNIX_FILE_SEPARATOR;
431 
432         for ( String file : files )
433         {
434             spec.print( "ln -s " );
435             spec.print( targetPrefix + file );
436             spec.print( " $RPM_BUILD_ROOT/" );
437             spec.println( sourcePrefix + file );
438 
439             linkSource.getSourceMapping().addLinkedFileNameRelativeToDestination( file );
440         }
441     }

443     /**
444      * {@link DirectoryScanner#scan() Scans} the <i>buildSourceLocation</i> using the
445      * {@link SoftlinkSource#getIncludes()} and {@link SoftlinkSource#getExcludes()} from <i>linkSource</i>. Returns the
446      * {@link DirectoryScanner} used for scanning.
447      *
448      * @param linkSource Source
449      * @param buildSourceLocation Build location where content exists.
450      * @return {@link DirectoryScanner} used for scanning.
451      */
452     private static DirectoryScanner scanLinkSource( final SoftlinkSource linkSource, final File buildSourceLocation )
453     {
454         final DirectoryScanner scanner = new DirectoryScanner();
455         scanner.setBasedir( buildSourceLocation );
456         List<String> includes = linkSource.getIncludes();
457         scanner.setIncludes( ( includes == null || includes.isEmpty() ) ? null
458                         : (String[]) includes.toArray( new String[includes.size()] ) );
459         List<String> excludes = linkSource.getExcludes();
460         scanner.setExcludes( ( excludes == null || excludes.isEmpty() ) ? null
461                         : (String[]) excludes.toArray( new String[excludes.size()] ) );
462         scanner.scan();
463         return scanner;
464     }

466     /**
467      * Assemble the RPM SPEC default file attributes.
468      *
469      * @return The attribute string for the SPEC file.
470      */
471     private String getDefAttrString()
472     {
473         final String defaultFilemode = mojo.getDefaultFilemode();
474         final String defaultUsername = mojo.getDefaultUsername();
475         final String defaultGroupname = mojo.getDefaultGroupname();
476         final String defaultDirmode = mojo.getDefaultDirmode();
477 
478         /* do not include %defattr if no default attributes are specified */
479         if ( defaultFilemode == null && defaultUsername == null && defaultGroupname == null
480             && mojo.getDefaultDirmode() == null && defaultDirmode == null )
481         {
482             return "";
483         }
484 
485         StringBuilder sb = new StringBuilder();
486 
487         if ( defaultFilemode != null )
488         {
489             sb.append( "%defattr(" ).append( defaultFilemode ).append( "," );
490         }
491         else
492         {
493             sb.append( "%defattr(-," );
494         }
495 
496         if ( defaultUsername != null )
497         {
498             sb.append( defaultUsername ).append( "," );
499         }
500         else
501         {
502             sb.append( "-," );
503         }
504 
505         if ( defaultGroupname != null )
506         {
507             sb.append( defaultGroupname ).append( "," );
508         }
509         else
510         {
511             sb.append( "-," );
512         }
513 
514         if ( defaultDirmode != null )
515         {
516             sb.append( defaultDirmode ).append( ")" );
517         }
518         else
519         {
520             sb.append( "-)" );
521         }
522 
523         return sb.toString();
524     }

526     /**
527      * Writes soft link from <i>linkSource</i> to <i>directory</i> using optional
528      * {@link SoftlinkSource#getDestination()} as the name of the link in <i>directory</i> if present.
529      *
530      * @param directory Directory to link to.
531      * @param linkSource Source to link from.
532      */
533     private void linkSingleFile( String directory, final SoftlinkSource linkSource )
534     {
535         spec.print( "ln -s " );
536         spec.print( linkSource.getLocation() );
537         spec.print( " $RPM_BUILD_ROOT/" );
538         spec.print( directory );
539         spec.print( '/' );
540         final String destination = linkSource.getDestination();
541         final String linkedFileName =
542             destination == null ? new File( linkSource.getMacroEvaluatedLocation() ).getName() : destination;
543         spec.println( linkedFileName );
544 
545         linkSource.getSourceMapping().addLinkedFileNameRelativeToDestination( linkedFileName );
546     }

548     /**
549      * Writes all the scriptlets to the <i>spec</i>.
550      */
551     private void writeScripts()
552         throws IOException
553     {
554         // all scriptlets in order to write
555         final Scriptlet[] scriptlets =
556             new Scriptlet[] { mojo.getPrepareScriptlet(), mojo.getPretransScriptlet(), mojo.getPreinstallScriptlet(),
557                 mojo.getPostinstallScriptlet(), mojo.getPreremoveScriptlet(), mojo.getPostremoveScriptlet(),
558                 mojo.getPosttransScriptlet(), mojo.getVerifyScriptlet(), mojo.getCleanScriptlet() };
559 
560         // all directives, in parallel to scriptlets
561         final String[] directives = new String[] { "%prep", "%pretrans", "%pre", "%post", "%preun", "%postun",
562             "%posttrans", "%verifyscript", "%clean" };
563 
564         for ( int i = 0; i < scriptlets.length; ++i )
565         {
566             if ( scriptlets[i] != null )
567             {
568                 scriptlets[i].write( spec, directives[i] );
569             }
570         }
571     }

573     /**
574      * If<i>value</i> is not <code>null</code>, writes the <i>value</i> to <i>spec</i>.
575      * <p>
576      * Writes in format: <code><i>directive</i>: <i>value</i></code>
577      * </p>
578      *
579      * @param directive
580      * @param value
581      */
582     private void writeNonNullDirective( final String directive, final String value )
583     {
584         if ( value != null )
585         {
586             spec.print( directive );
587             spec.print( ": " );
588             spec.println( value );
589         }
590     }

592     /**
593      * Writes a new line for each element in <i>strings</i> to the <i>writer</i> with the <i>prefix</i>.
594      *
595      * @param strings <tt>List</tt> of <tt>String</tt>s to write.
596      * @param prefix Prefix to write on each line before the string.
597      */
598     private void writeList( Collection<String> strings, String prefix )
599     {
600         if ( strings != null )
601         {
602             for ( String str : strings )
603             {
604 
605                 if ( !str.equals( "" ) )
606                 {
607                     spec.print( prefix );
608                     spec.println( str );
609                 }
610             }
611         }
612     }
